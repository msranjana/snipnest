  description: "Capitalizes the first letter of each word in a string, also known as title casing.",
  keywords: ["string", "casing", "title-case"],

};

```javascript
function capitalizeWords(str) {
  return str.replace(/\b\w/g, (char) => char.toUpperCase());
}
```

```javascript
capitalizeWords("hello world"); // "Hello World"
```
